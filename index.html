import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", }; Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: corsHeaders }); const body = await req.text(); console.log("Body recebido:", body); try { const { relatorio_id, doc_id } = JSON.parse(body); console.log("relatorio_id:", relatorio_id); console.log("doc_id:", doc_id); if (!doc_id) throw new Error("doc_id não informado"); const token = Deno.env.get("AUTENTIQUE_TOKEN")!; console.log("Token disponível:", !!token); const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); const query = `query { document(id: "${doc_id}") { id name files { signed } signatures { public_id name email } } }`; const res = await fetch("https://api.autentique.com.br/v2/graphql", { method: "POST", headers: { "Authorization": `Bearer ${token}`, "Content-Type": "application/json", }, body: JSON.stringify({ query }), }); const text = await res.text(); console.log("Resposta Autentique:", text); let data: any; try { data = JSON.parse(text); } catch { throw new Error(`JSON inválido: ${text.substring(0,200)}`); } if (data.errors) throw new Error(JSON.stringify(data.errors)); const doc = data.data?.document; if (!doc) throw new Error("Documento não encontrado no Autentique"); console.log("PDF signed:", doc.files?.signed); console.log("Signatures:", JSON.stringify(doc.signatures)); const temPdf = !!(doc.files?.signed); if (temPdf) { await supabase.from("relatorios_vistoria").update({ status: "assinado", docusign_status: "completed", pdf_url: doc.files.signed, }).eq("id", relatorio_id); await supabase.from("relatorios_vistoria_audit").insert({ relatorio_id, alterado_por_email: "sincronizacao-manual", campo_alterado: "status", valor_anterior: "enviado", valor_novo: "assinado — PDF disponível", }); } return new Response(JSON.stringify({ ok: true, assinado: temPdf, pdf_url: doc.files?.signed || null, signatarios: doc.signatures, }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, }); } catch (e) { console.error("ERRO:", String(e)); return new Response(JSON.stringify({ error: String(e) }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" }, }); } });
